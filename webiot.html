Available networks 
#include <WiFi.h>

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(100);
}

void loop() {
  int n = WiFi.scanNetworks();
  Serial.println("Available Networks:");
  for (int i = 0; i < n; i++) {
    Serial.print(i + 1);
    Serial.print(" ");
    Serial.print(WiFi.SSID(i));
    Serial.print(" Signal = ");
    Serial.print(WiFi.RSSI(i));
    Serial.println(" dBm");
  }
  delay(10000);
}
